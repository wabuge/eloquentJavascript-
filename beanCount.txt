

function countBs(a){
    let count = 0;
   for( let i=0; i<a.length; i++){
       if( a[i] == "B"){
        count ++}
   } return count ++
}
console.log(countBs("BBC"))

function countChar(x,y){
    let count= 0;
    for(let i=0; i<x.length; i++){
        if(x[i]== y)
        count ++ 
    }
    return count ++

}
console.log(countChar("kkakkar", "k"))
